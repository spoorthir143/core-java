			}

	{
		boolean isUpdated=false;

				sportsNames[index]=updatedSport;

5)Write a java program using parameterized constructor?
class Employee {

    int id;
    String name;
    double salary;

    Employee(int i, String n, double s) {
        id = i;
        name = n;
        salary = s;
    }

    void display() {
        System.out.println("ID: " + id);
        System.out.println("Name: " + name);
        System.out.println("Salary: " + salary);
       
    }

    public static void main(String[] args) {
        Employee e1 = new Employee(1, "Spoorthi", 25000);
        Employee e2 = new Employee(2, "Rahul", 30000);

        e1.display();
        e2.display();
    }
}
